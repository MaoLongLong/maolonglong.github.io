Go 函数选项模式（Functional Options）

2021年7月3日
| Golang

在 Python 中经常可以看到这样的代码：

def sub(x=3, y=1):

函数可以设置参数默认值。并且调用时，能指定名称以自由调换参数位置。

go 中显然没有这样的语法糖，但是我们能通过函数选项模式实现类似的功能。

package main

import "fmt"

// 将参数封装成结构体
type Args struct {
	StringArg string
	IntArg    int
}

// 参数默认值
var defaultArgs = Args{
	StringArg: "default string",
	IntArg:    666,
}

type Option func(p *Args)

func WithStringArg(s string) Option {
	return func(p *Args) {
		p.StringArg = s
	}
}

func WithIntArg(i int) Option {
	return func(p *Args) {
		p.IntArg = i
	}
}

func Method(opt ...Option) {
	args := defaultArgs
	for _, o := range opt {
		o(&args)
	}
	fmt.Printf("%#v\n", args)
}

func main() {
	// 全部使用默认值
	Method()

	// 全部指定
	Method(WithStringArg("hello"), WithIntArg(18))

	// 仅指定 int arg
	Method(WithIntArg(19))
}
